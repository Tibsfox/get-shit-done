  <name>Task 1: Create Logger class with level methods and session tracking</name>
  <files>lib/logger.js</files>
  <action>
Create lib/logger.js:

1. Import dependencies:
   ```javascript
   const { randomUUID } = require('node:crypto');
   const { loadConfig, LEVELS, LEVEL_NAMES } = require('./logger-config');
   const { SyslogTransport, GSD_TO_SEVERITY } = require('./logger-syslog');
   ```

2. Logger class:
   ```javascript
   class Logger {
     constructor(options = {}) {
       this.config = options.config || loadConfig();
       this.level = this.config.level;
       this.category = options.category || 'gsd';
       this.sessionId = options.sessionId || randomUUID();
       this.transport = this._createTransport();
     }
   }
   ```

3. _createTransport() method:
   - If syslog disabled in config, return null
   - Create SyslogTransport with config.syslog options
   - Set appName to 'gsd'

4. _shouldLog(level) method:
   - Return false if this.level === 0 (OFF)
   - Return false if level > this.level
   - Return true otherwise
   - This is the short-circuit check per LEVEL-01

5. _log(level, message, context = {}) method:
   - if (!this._shouldLog(level)) return; // Short-circuit
   - Build structured data: { session: this.sessionId, category: this.category, ...context }
   - Get syslog severity from GSD_TO_SEVERITY[level]
   - If transport exists and severity !== null, call transport.send()
   - Also log to console if config.console is true (for development)

6. Level-specific methods (call _log with appropriate level):
   - error(message, context) -> _log(LEVELS.ERROR, message, context)
   - warn(message, context) -> _log(LEVELS.WARN, message, context)
   - info(message, context) -> _log(LEVELS.INFO, message, context)
   - debug(message, context) -> _log(LEVELS.DEBUG, message, context)
   - trace(message, context) -> _log(LEVELS.TRACE, message, context)

7. log(level, message, context) generic method:
   - Accept level as number or string
   - Convert string to number via LEVELS[level.toUpperCase()]
   - Call _log(numericLevel, message, context)
  </action>

Expected: Message sent to syslog (or graceful no-op if syslog unavailable)
  </verify>
  <done>Logger class logs at all levels, session ID included, short-circuit works for OFF level</done>

   ```javascript
   let instance = null;

   function createLogger(options = {}) {
     if (!instance) {
       instance = new Logger(options);
     }
     return instance;
   }

   function getLogger() {
     return instance || createLogger();
   }

   function resetLogger() {
     instance = null; // For testing
   }
   ```
